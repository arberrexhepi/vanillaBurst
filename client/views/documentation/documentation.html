//naming convention `${view}Request` see schema>appConfigSample.js for detailed explanation

    //view roll call, specifying the order in which to load functions, great for debugging, you can comment out a view name to see what is causing potential issues, super control!  

    window.appRequest = async function appRequest(renderSchema, navPackage, runFunction) {

      runFunction = window.runFunction;

      await runFunction; //this is a special case of runFunction that isn't auto set and is designed to respect runRoll's value (see further below)

//there may be custom calls to this request file other than strictly from initial runtime to this function, and the window.runFunction needs to be set to 'functionBurst' to reRoll this chain.

      navPackage =    window.modalPackage //this is for if you are bringing additional view functions, set in globals>config.js as a string or array of parts called in window.routeParts JS object. This will be explained in globals>configSample.js

       renderSchema = window.renderSchema;

       if(runFunction){

       rollCall = [

        'app',

        'somefeature',

        'myposts',

         ...navPackage, //this can be a nested `${someotherView}Request` which is treated as a chained promise. A must for any functionality that loads additional data or serverResult will return the last data retreived

//ie you've defined a schema config for navigation for; schemas>navConfig.js, and schemas>subNav.js ... and loaded by adding to window.routeParts() object, setting them to an array for spread use ...navPackage

       ];

 

       runRoll = "rollBurst"; //this controlls whether the chain should run. This isn't global, and you set it per Request. You can use it to turn off the entire Request chain for debugging. setting it to rollBurst, initiates it

       window.rollCall = rollCall; makes this rollCall available to window, no pollution, good to go here!

         await window.childFunction(renderSchema, rollCall, runRoll); // running the promises

       console.log("shopRequest is sending rollCall "+rollCall); //although not ideal you could continue to run client logic functions here. However the best use case here is to use the remainder of the scope to test for variables...anything in the config will be available here provided you call it properly via renderSchema

      }

    }

//sharedParts is the return of other object spread functions, this will be explained in configSample.js

    window.appConfig = function appConfig(sharedParts){

        let appConfig = { //the object name must match the file, which matches the function

            'landing': 'app', //the view name (same as view file name) ie this is a reference to the view client>views>app>app.js

            'scripts': window.vanillaBurstScripts, //spreads array of framework required files, will move this logic to framework in future implementations to reduce rewrite of this, possibly leaving only the customFunctions required by developers to be set

            'preloader': baseUrl + 'preloader.js', //baseUrl must be set in globals>config.js, this file comes with basic loader but can be edited

            'customFunctions': {//files build for a complete feature set of view.

              'appRequest': { // naming is `${view}Request`

                'role': 'rollCall', //a rollCall role function is available for `${view}Request` files only. This is a file with function where you can control the order of loading procedure for granual control of runtime, although a decent ordering in this file goes a long way for readability

                'dir': 'server/api/app/', //the directory of this file (baseUrl is prefixed by framework) all 1${view}Request` files are located in that folder. The nested structure may feel over-nested until your app grows to many parts! Customization beyound the .../app/ scope is possible as you can chain requests to respect promises

                'functionFile': 'shopRequest', //this defines the name of the function, however it is named this way because it is easily identifiable, and is an extended variable via renderSchema.customFunctions.app.functionFile (renderSchema returns the complete `${view}Config` config, and is accessible to all the functions of this function without interferrance from other configs. who said promises are bad.. that's just insane lol

                'render': 'burst', //this flag alerts childFunction.js that this must be immediately sent through the chain as soon as the `${view}Request` file calls window.childFunction(renderSchema, rollCall, runFunction), this will be explained in 'appRequestSample.js'

                'originBurst':'shop'

              },

//end of request file

              'app': {

                'role': 'parent',

                'dir': 'client/views/app/',

                'functionFile': 'app',

                'render': 'pause', //flagged to not load immediately, awaits promises

                'originBurst':'app' //originBurst at runtime becomes an invaluable state manager for anything you want to maintain about a certain function... callable via window.originBurst[view], and for data calling functions makes window.originBurst[view].serverResult available

              },

              'somefeature': {

                'dir': 'client/views/app/anyname/',

                'functionFile': 'somefeature',

                'render': 'pause',

                'originBurst':'app'

              },

       

              ...sharedParts, //spreads reusable parts, where it's originBurst param = this view, auto set

              'myposts': {

                'dir': 'client/views/app/posts/',

                'functionFile': 'myposts',

                'render': 'pause', //flagged to not load immediately, awaits promises

                'originBurst':'app', //originBurst at runtime becomes an invaluable state manager for anything you want to maintain about a certain function... callable via window.originBurst[view], and for data calling functions makes window.originBurst[view].serverResult available

                'dataSchema': { //this will build data when this function is called and you can use its result immediately when runFunction === 'functionBurst' with window.getData(view, buildView) where buildView(viewResult) is your callback to run functions.  This will be explained in appSample.js

                  'returnResult': 'true', // returnResult true brings this serverResult of `${view}Result` to your window.view(renderSchema, serverResult, runFunction, originBurst) call, and sets it to window.originBurst.view.serverResult for further interaction, ie moving around state changes

                  'resultTarget': 'app', //sending the result to a namespace other than the current function. Dangerous but possible! Default it to view name

                  'url': undefined, // for dynamically set dataSchema, set as undefined, meaning it won't return data at initial call. Later when requested, this object can be built via renderSchema.customFunctions.view.dataSchema, different from the maluable originBurst, we set it here, incase that data is removed, when a state is recalled, it defaults to your latest call. Neat!

                  'method': 'POST', //set your method

                  'data':  undefined, //can be build based on parameters

                    '_ajax_nonce': wp_ajax.nonce, // custom params for security purposes or additional fetch header options like redirect, etc, etc in this case wordpress ajax nonce has been defined

                 

                },

              },

             

            },

          }

          return shopConfig; //this is what returns the object when it is called in schema.js

    }
